Habit Tracker (Python)


from datetime import datetime

class Habit:
    def __init__(self, name, date, done=False): 
        self.name = name
        self.date = date
        self.done = done

    def mark_complete(self):
        self.done = True

    def __str__(self):
        status = "✓" if self.done else "✗"
        return f"{self.date}:{self.name} [{status}]"
